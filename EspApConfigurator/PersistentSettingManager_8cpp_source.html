133 uint8_t PersistentSettingManager::calculateSignature()
134 {
135  // XOR bysubSig of id, address & typecode for all settings
136  uint8_t signature = 0;
137  for (uint8_t settingIndex=0; settingIndex<_count; settingIndex++) {
138  uint8_t subSig = 0;
139  uint8_t i;
140  // add setting name
141  for (i=0; i<_settings[settingIndex].id.length(); i++) {
142  subSig ^= _settings[settingIndex].id[i];
143  }
144 
145  // add setting type
146  for (i=0; i<_settings[settingIndex].setting->typecode().length(); i++) {
147  subSig ^= _settings[settingIndex].setting->typecode()[i];
148  }
149 
150  // add setting address
151  uint16_t address = _settings[settingIndex].setting->address();
152  subSig ^= ((uint8_t*)&address)[0];
153  subSig ^= ((uint8_t*)&address)[1];
154 
155  // the sig for this setting is multipled by it's index. This way if
156  // settings are just re-ordered, that will trigger a sig fail
157  signature ^= (subSig * settingIndex) % 256;
158  }
159  return signature;
160 }
